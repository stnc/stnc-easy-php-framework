Array To String

array olarak gelen veriyi string olarak geri dondurur

			
        $arr = new \Lib\Arrays();
        
        $arrayExpression = array(
            "marka",
            "hasan",
            "veli"
        );
       
        echo $arr->arrayToString($arrayExpression);
