Tools Sınıfı

Araçlar sınıfı sürekli olarak kullanıclak araçlar burada yer alacak

printr fonksiyonu

print_r nin pre tag li olan çıktısı 
			
        $tool = new \Lib\Tools();
        $data = array(
            'UrunID' => 02,
            'UrunAdi' => "çikolata  ",
            'Resim' => "biskuvi.jpg",
            'ResimURL' => "biskuvi.jpg",
            'URL' => "biskuvi.jpg",
            'Fiyat' => 40.99,
            "ToplamAdet" => 1,
            "ToplamFiyat" => ""
        );
        $tool->printr($data);

is_ajax fonksiyonu

isteğin ajax dan olup olmadığını kontrol eder 
			
        $tool = new \Lib\Tools();
        $tool->is_ajax();

getRealIpAddr() fonksiyonu

kişinin gerçek ip adresini verir 
			
        $tool = new \Lib\Tools();
        $tool->getRealIpAddr();

currentPageURL() fonksiyonu

sayfanın aktif url adresini verir 
			
        $tool = new \Lib\Tools();
        $tool->currentPageURL();

curPageName() fonksiyonu

sayfa ismini verir 
			
        $tool = new \Lib\Tools();
        $tool->curPageName();
        //sonuc index.php
