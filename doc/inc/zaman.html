aylar fonksiyonu

Select listlerde ayları döndürmek içindir

	   
        $zaman = new \Lib\ZamanOlaylari();
        print_r ($zaman->Aylar() );      
		//çıktısı 
		Array
		(
		    [01] => Ocak
		    [02] => Şubat
		    [03] => Mart
		    [04] => Nisan
		    [05] => Mayıs
		    [06] => Haziran
		    [07] => Temmuz
		    [08] => Ağustos
		    [09] => Eylül
		    [10] => Ekim
		    [11] => Kasım
		    [12] => Aralık
		)

hangi_ay_tr fonksiyonu

ayın sayı değerini yazı olarak verir 

	   
        $zaman = new \Lib\ZamanOlaylari();
     echo $zaman->hangi_ay_tr ("02");
     // sonuc şubat

hangi_gun_tr fonksiyonu

ingilizce gunun türkçe karşılığını verir 
	   
        $zaman = new \Lib\ZamanOlaylari();
     echo $zaman->hangi_gun_tr("Monday"); 
     //sonuc pazartesi

hangi_gun_tr fonksiyonu

   	   $zaman = new \Lib\ZamanOlaylari();
        
        echo $zaman->kacGunsonrasi(2);

sql2tr_Date fonksiyonu

veritabnındaki msql formatlı tarihi türkiye tipine cevirir 
	   

        echo $zaman->sql2tr_Date("2014-04-25");
        // sonuc 25-04-2014

tr2sql_DateTimefonksiyonu

Türkiye tarihini veritabanındaki mysql formatlı tipe cevirir 
	   

       $zaman = new \Lib\ZamanOlaylari();
        
        echo $zaman->tr2sql_DateTime("25-04-2014 15:40:15"); 
        //sonuc 2014-04-25 15:40:15

sql2tr_DateTime_GunAy

veritabnındaki msql formatlı tarihi türkiye tipine yazılı olarak cevirir 
	   

           $zaman = new \Lib\ZamanOlaylari();
        echo $zaman->sql2tr_DateTime_GunAy("2015-08-29 11:00:15");
   //     sonuc: 29 Ağustos 2015 Pazartesi 11:00
